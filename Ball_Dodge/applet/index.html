Problem: Compute and animate free fall motion of missles that hits the target after 2 seconds.
Solution: using the formula for projectile motion:
Position(time) = time squared / 2 * Gravity + Velocity(time) + initial Position,
apply a velocity and acceleration (gravity) to the missles being launched at the target.

Pseudocode:

class Spring { //reference http://processing.org/learning/topics/springs.html
void update() {
//creates inertia for the target ball ASSIGN
resting position of Y = mouseY;
resting position of X = mouseX;
COMPUTE
force = -k * y; //f = -ky
acceleration = force / mass; //set the acceleration f=ma == a=f/m
velocity of y = damp * (velocity of y + acceleration); //set the velocity
temporary y position = temporary y position + velocity y; //update position
repeat for x's
}
}

class BALL {
void move(float time, point Target, int i) {
if frame >= 120 {
ASSIGN
Center point to Home point;
Velocity = aim vector;
reset frame;
}
else {
REASSIGN
Velocity y component = Velocity y component + time*Gravity y component; //apply gravity acceleration
Center x component = Center x component + time*Velocity x component;
Center y component = Center y component + 1/2*Gravity y component*time squared + time*Velocity y component;
}
}
}

pt predict() { //predict target location and aim there 2 seconds late
SET
point TargetCenter = P(Target temporary x position, Target temporary Y position);
point previousTargetCenter = P(mouseX, mouseY);
vector PREDICT = vector(TC, pTC);
}

vec aim(Home point, Target point, int i) { //apply gravity to the ball's trajectory
if ball shooting from right corner {
COMPUTE
Velocity x component = (Target x component - width)*(1/2);
Velocity y component = (Target y component - height - 2*(Gravity y component))*(1/2);
initial Velocity = vector(Velocity x and y);
}
reverse for left corner

processCollisions() {
point TargetCenter = P(Target temporary x position, Target temporary Y position);
for each missle {
if the distance between the Target Center and Missle Center is less than the diameter of the ball {
add to points counter;
change boolean value to check collision to TRUE;
change color of missile;
}
if they are not colliding {
change boolean value to check collision to FALSE;
change color back to pink;
}
}
}

How-to-play:
Move cursor to move the green target ball around.
When you hit a pink ball, you get a point!
Collect points until you get tired.
